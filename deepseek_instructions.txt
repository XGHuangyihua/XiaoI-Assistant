```python
#Program start
import pyautogui
import time
import pyperclip
import win32api

def main():
    try:
        import win32api
        win32api.LoadKeyboardLayout('00000409', 1)
        time.sleep(0.3)
    except ImportError:
        pass
    try:
        print("Opening Run dialog...")
        pyautogui.hotkey('win', 'r')
        time.sleep(0.5)
        print("Typing calc...")
        pyautogui.write('calc')
        time.sleep(0.3)
        pyautogui.press('enter')
        time.sleep(2)
        return {"success": True}
    except Exception as e:
        return {"success": False, "error": str(e)}

if __name__ == "__main__":
    result = main()
    print(f"Execution result: {result}")
#Program end
```
