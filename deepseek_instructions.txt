   - pyautogui.moveTo(x, y) 移动鼠标

【示例 - 打开计算器】
```python
#Program start
import pyautogui
import time
import pyperclip
import ctypes

def main():
    try:
        # 切换到英文输入法（模拟 Alt+Shift）
        KEYEVENTF_KEYDOWN = 0x0000
        KEYEVENTF_KEYUP = 0x0002
        ctypes.windll.user32.keybd_event(0x12, 0, KEYEVENTF_KEYDOWN, 0)
        time.sleep(0.1)
        ctypes.windll.user32.keybd_event(0x10, 0, KEYEVENTF_KEYDOWN, 0)
        time.sleep(0.1)
        ctypes.windll.user32.keybd_event(0x10, 0, KEYEVENTF_KEYUP, 0)
        time.sleep(0.1)
        ctypes.windll.user32.keybd_event(0x12, 0, KEYEVENTF_KEYUP, 0)
        time.sleep(0.3)
        
        print("打开运行对话框...")
        pyautogui.hotkey('win', 'r')
        time.sleep(0.5)
        print("输入 calc...")
        pyautogui.write('calc')
        time.sleep(0.3)
        pyautogui.press('enter')
        time.sleep(2)
        return {"success": True}
    except Exception as e:
        return {"success": False, "error": str(e)}

if __name__ == "__main__":
    result = main()
    print(f"执行结果: {result}")
#Program end
```
